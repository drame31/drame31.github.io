ARCHITECTURE
How It's Built

Python Flask REST API

  • Flask backend on port 5000
  • SQLite database with sqlite3
  • CORS enabled for browser access
  • Full CRUD: GET / POST / PUT / DELETE
  • Auto DB init on first run

JavaScript Frontend

  • Auto-detects local Python API
  • Falls back to localStorage (demo)
  • Drag & drop with HTML5 Drag API
  • Kanban + List views
  • Real-time search & filters

SQLite Schema

  • jobs(id, company, role, status)
  • url, notes, priority fields
  • created_at with auto timestamp
  • Lightweight, zero config DB

Run Locally

  • pip install flask flask-cors
  • python api.py
  • Open hireflow.html in browser
  • Badge switches from DEMO → LIVE
api.py — Flask REST API 
from flask import Flask, jsonify, request
from flask_cors import CORS
import sqlite3
from datetime import datetime

app = Flask(__name__)
CORS(app)

DB = 'hireflow.db'

# ── DB helper ──────────────────────────────
def get_db():
    conn = sqlite3.connect(DB)
    conn.row_factory = sqlite3.Row
    return conn

def init_db():
    db = get_db()
    db.execute('''CREATE TABLE IF NOT EXISTS jobs (
        id        INTEGER PRIMARY KEY AUTOINCREMENT,
        company   TEXT NOT NULL,
        role      TEXT NOT NULL,
        status    TEXT DEFAULT 'applied',
        url       TEXT,
        notes     TEXT,
        priority  TEXT DEFAULT 'medium',
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )''')
    db.commit()

# ── Routes ─────────────────────────────────
@app.route('/api/jobs', methods=['GET'])
def get_jobs():
    db = get_db()
    jobs = db.execute(
        'SELECT * FROM jobs ORDER BY created_at DESC'
    ).fetchall()
    return jsonify([dict(j) for j in jobs])

@app.route('/api/jobs', methods=['POST'])
def create_job():
    data = request.json
    db = get_db()
    db.execute(
        'INSERT INTO jobs (company,role,status,url,notes,priority) VALUES (?,?,?,?,?,?)',
        [data['company'], data['role'], data.get('status','applied'),
         data.get('url',''), data.get('notes',''), data.get('priority','medium')]
    )
    db.commit()
    return jsonify({'success': True}), 201

@app.route('/api/jobs/<int:jid>', methods=['PUT'])
def update_job(jid):
    data = request.json
    db = get_db()
    db.execute(
        'UPDATE jobs SET company=?,role=?,status=?,url=?,notes=?,priority=? WHERE id=?',
        [data['company'], data['role'], data['status'],
         data.get('url',''), data.get('notes',''), data.get('priority','medium'), jid]
    )
    db.commit()
    return jsonify({'success': True})

@app.route('/api/jobs/<int:jid>', methods=['DELETE'])
def delete_job(jid):
    db = get_db()
    db.execute('DELETE FROM jobs WHERE id=?', [jid])
    db.commit()
    return jsonify({'success': True})

@app.route('/api/stats')
def stats():
    db = get_db()
    rows = db.execute('SELECT status, COUNT(*) as n FROM jobs GROUP BY status').fetchall()
    return jsonify({r['status']: r['n'] for r in rows})

# ── Start ───────────────────────────────────
if __name__ == '__main__':
    init_db()
    print('HireFlow API → http://localhost:5000')
    app.run(debug=True, port=5000)
